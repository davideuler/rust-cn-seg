POST /api/segment 中文分词

请求体 Request Body

{ "text": "我来到北京清华大学", "mode": "default" }

响应 Response

{ "words": ["我", "来到", "北京", "清华大学"], "count": 4, "elapsed_ms": 0.08 }
curl
Python
TypeScript
Rust
curl -X POST http://localhost:3001/api/segment \
  -H "Content-Type: application/json" \
  -d '{"text": "我来到北京清华大学", "mode": "default"}'
import requests

response = requests.post(
    "http://localhost:3001/api/segment",
    json={"text": "我来到北京清华大学", "mode": "default"}
)
result = response.json()
print(result["words"])      # ['我', '来到', '北京', '清华大学']
print(result["elapsed_ms"])  # e.g. 0.08
const response = await fetch("http://localhost:3001/api/segment", {
  method: "POST",
  headers: { "Content-Type": "application/json" },
  body: JSON.stringify({ text: "我来到北京清华大学", mode: "default" }),
});
const result = await response.json();
console.log(result.words);      // ['我', '来到', '北京', '清华大学']
console.log(result.elapsed_ms); // 0.08
use reqwest::Client;
use serde_json::json;

#[tokio::main]
async fn main() -> Result<(), Box<dyn std::error::Error>> {
    let client = Client::new();
    let resp = client
        .post("http://localhost:3001/api/segment")
        .json(&json!({"text": "我来到北京清华大学", "mode": "default"}))
        .send()
        .await?
        .json::<serde_json::Value>()
        .await?;
    println!("{:?}", resp["words"]);
    Ok(())
}
